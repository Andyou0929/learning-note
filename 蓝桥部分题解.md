    }

        handleStatic(res, `..${req.url}`, 'js');
    }

    if (req.method === 'GET' && req.url === '/users') {
        // TODO 处理获取文件内容的操作
        let fileContent = '';
        fs.readFile(dataUrl,"utf-8",(err,str)=>{
            send(res,0,"",eval(str))
        })
    }
    else if (req.method === 'PUT' && req.url === '/editUser') {
        let body = '';
        req.on('readable', () => {
            let chunk = '';
            if (null !== (chunk = req.read())) {
                body += chunk;
            }
        })
        req.on('end', () => {
            if (body) {
                // TODO 处理更改文件数据并将最新的文件数据响应给客户端
                body = JSON.parse(body)
                fs.readFile(dataUrl,"utf-8",(err,str)=>{
                    str = JSON.parse(str)
                    str.forEach(item => {
                        if (item.id === body.params.id) {
                            item.power = body.data.power
                        }
                    });
                    send(res,0,"",str)
                    fs.writeFile(dataUrl,JSON.stringify(str),()=>{})
                })
            }
        })
    }
    else if (req.method === 'POST' && req.url === '/logger') {
        let body = '';
        req.on('readable', () => {
            let chunk = '';
            if (null !== (chunk = req.read())) {
                body += chunk;
            }
        })
        req.on('end', () => {
            if (body) {
                // TODO 处理新增日志
                let obj = {
                    id: getLoggerId(), //日志的id
                    msg: JSON.parse(body).data, //日志信息（请求体数据）
                    time: nowDate(), //创建日志的当前时间
                }
                fs.readFile(loggerUrl,"utf-8",(err,str)=>{
                    str = str ? JSON.parse(str) : []
                    str.push(obj)
                    let inputStr = "[\r"
                    str.forEach((item,index)=>{
                        inputStr += (JSON.stringify(item) + `${(str.length -1) === index ? "" : ","}\r`)
                    })
                    inputStr += "]"
                    send(res,0,"",str.reverse())
                    fs.writeFile(loggerUrl,inputStr,()=>{})
                })
            }
        })
    }
})
function nowDate() {

    let month = date.getMonth() + 1
    let day = date.getDate()
    let hours = date.getHours()
    let minutes = date.getMinutes()
    let seconds = date.getSeconds()
    return `${year}/${month}/${day} ${hours <= 12 ? "上午" : "下午"}${hours <= 12 ? hours : (hours%12)}:${minutes}:${seconds}`
}

let generateData = (data) => {
  let reg1 = /^\{\{(repeat\([\d,\s]+\))\}\}$/
  let reg2 = /^\{\{bool\(\)\}\}$/
  let reg3 = /^\{\{integer\([\d,\s]+\)\}\}$/
  if (!Array.isArray(data)) {
    for (let k in data){
      if(reg2.test(data[k]) || reg3.test(data[k])) data[k] = eval(data[k])
    }
    return data
  }
  let len = reg1.test(data[0]) ? eval(data[0]) : 1
  let result = []
  while (result.length !== len) {
    let obj = {...data[1]}
    for (let k in obj) {
      if(reg2.test(obj[k]) || reg3.test(obj[k])) obj[k] = eval(obj[k])
    }
    result.push(obj)
  }
  function repeat(a,b){
    return  b ? Math.random() * (b-a+1) + a | 0 : a
  }
  function integer(a, b) {
    return Math.random() * (b - a + 1) + a | 0
  }
  function bool() {
    return Math.random() > 0.5
  }
  return result
};
